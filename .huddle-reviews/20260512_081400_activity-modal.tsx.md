function formatRelative(iso: string): string {
  const diff = Date.now() - new Date(iso).getTime();
  const min = Math.floor(diff / 60000);
  if (min < 1) return "たった今";
  if (min < 60) return `${min}分前`;
  const hr = Math.floor(min / 60);
  if (hr < 24) return `${hr}時間前`;
  const day = Math.floor(hr / 24);
  if (day < 7) return `${day}日前`;
  return new Date(iso).toLocaleDateString("ja-JP", { month: "short", day: "numeric" });
}

function previewContent(content: string): string {
  const firstLine =
    content.split("\n").find((l) => l.trim().length > 0 && !l.startsWith("https://")) || "";
  return firstLine.length > 40 ? firstLine.slice(0, 40) + "…" : firstLine;
}

function Avatar({ url, name }: { url: string | null; name: string }) {
  if (url) {
    /* eslint-disable-next-line @next/next/no-img-element */
    return (
      <img
        src={url}
        alt={name}
        className="w-9 h-9 rounded-full object-cover shrink-0 mt-0.5"
      />
    );
  }
  return (
    <div className="w-9 h-9 rounded-full bg-accent/20 flex items-center justify-center shrink-0 mt-0.5">
      <span className="text-xs font-bold text-accent">{(name || "?")[0]?.toUpperCase()}</span>
    </div>
  );
}

## Codex レビュー

- [優先度: 今すぐ] `mark_activity_seen` / `mark_mention_seen` / `mark_reply_seen` に `workspaceId` が渡っていません。RPC がワークスペース単位で既読管理する仕様でない場合、別ワークスペースの未読まで既読化されます。取得系と同じく `p_workspace_id: workspaceId` を渡す形に揃えてください。

- [優先度: 今すぐ] `useEffect` 内で `cancelled` チェック後にも `await supabase.rpc(...)` と `setLoaded` / `setUnread` / `dispatchEvent` が続くため、アンマウント後やタブ切替後に古い処理が状態を更新できます。各 `await` の後、特に既読RPC後と最後の `setUnread` 前に `if (cancelled) return;` を入れてください。

- [優先度: 後で] Supabase RPC の `error` を無視しているため、取得失敗時もローディングが残る、または既読扱いだけ進む可能性があります。各RPCの `{ data, error }` を確認し、失敗時は `loaded` を更新するかエラー表示に落とすなど、UIが詰まらない処理を追加してください。

- [優先度: 後で] 未読内訳取得で `supabase.auth.getUser()` の結果が `undefined` でもRPCを呼んでいます。ユーザー未取得時は早期returnし、RPCには必ず有効な `user.id` を渡してください。
