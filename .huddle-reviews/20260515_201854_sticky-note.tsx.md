"use client";

import type { BoardNoteWithProfile } from "@/lib/supabase/types";

// カテゴリ名から背景色を決定（同じカテゴリは同じ色）
const CATEGORY_COLORS = [
  "bg-amber-100 border-amber-300",
  "bg-blue-100 border-blue-300",
  "bg-green-100 border-green-300",
  "bg-pink-100 border-pink-300",
  "bg-purple-100 border-purple-300",
  "bg-orange-100 border-orange-300",
  "bg-teal-100 border-teal-300",
  "bg-rose-100 border-rose-300",
];

function categoryColor(category: string | null): string {
  if (!category) return "bg-gray-100 border-gray-300";
  let hash = 0;
  for (let i = 0; i < category.length; i++) {
    hash = ((hash << 5) - hash + category.charCodeAt(i)) | 0;
  }
  return CATEGORY_COLORS[Math.abs(hash) % CATEGORY_COLORS.length];
}

function relativeTime(dateStr: string): string {
  const diff = Date.now() - new Date(dateStr).getTime();
  const min = Math.floor(diff / 60000);
  if (min < 1) return "たった今";
  if (min < 60) return `${min}分前`;
  const hr = Math.floor(min / 60);
  if (hr < 24) return `${hr}時間前`;
  return new Date(dateStr).toLocaleDateString("ja-JP", { month: "short", day: "numeric" });
}

type Props = {
  note: BoardNoteWithProfile;
};

export function StickyNote({ note }: Props) {
  const colorClass = categoryColor(note.category);

  const avatarUrl = note.profiles?.avatar_url;

## Codex レビュー

- [優先度: 今すぐ] `relativeTime()` が不正な `created_at` や未来日時を考慮していません。`new Date(dateStr).getTime()` が `NaN` の場合や `diff < 0` の場合に表示が壊れます。`Number.isNaN(timestamp)` なら空文字/固定文言、未来なら `たった今` などに丸めるガードを入れてください。

- [優先度: 今すぐ] `relativeTime(note.created_at)` を render 中に `Date.now()` で計算しているため、Next.js の初期HTMLとHydration時刻がずれて hydration mismatch になる可能性があります。`useEffect` 後に表示する、または `suppressHydrationWarning` を付けるなど、初期描画差分を避けてください。

- [優先度: 後で] 投稿者行の `displayName` は `truncate` だけだと flex 内で効かず、長い名前が時刻を押し出す可能性があります。親または名前側に `min-w-0` を付け、名前 span に `max-w-*` などを指定してください。

- [優先度: 後で] `relativeTime()` と `categoryColor()` は表示ロジックとして分離されているので、最低限の単体テストを追加してください。特に `null` カテゴリ、同一カテゴリの色安定性、不正日付、未来日付、24時間超の表示をカバーするとよいです。
