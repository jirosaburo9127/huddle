```
// ブラウザ通知を表示するユーティリティ
export function showMessageNotification({
  senderName,
  channelName,
  content,
  url,
}: {
  senderName: string;
  channelName: string;
  content: string;
  url: string;
}) {
  if (typeof window === "undefined") return;
  if (typeof Notification === "undefined") return;
  if (Notification.permission !== "granted") return;

  // 現在見ているチャンネルと同じなら、フォーカス中は通知しない
  // 別チャンネルのメッセージならフォーカス中でも通知する（LINE方式）
  // urlが ?m=xxx 付きでも、pathnameだけで同一チャンネル判定する
  const targetPath = new URL(url, window.location.origin).pathname;
  const isCurrentChannel = window.location.pathname === targetPath;
  if (isCurrentChannel && document.hasFocus()) return;

  // メッセージ本文を100文字に切り詰め
  const body = content.length > 100 ? content.slice(0, 100) + "…" : content;
  const title = `${senderName} (#${channelName})`;

  const notification = new Notification(title, {
    body,
    icon: "/icons/icon-192.png",
    tag: `huddle-${Date.now()}`,
  });

  // 通知クリックで該当投稿に遷移
  notification.onclick = () => {
    window.focus();
    if (url) {
      // 同一チャンネル内ならCustomEventでジャンプ、別チャンネルなら通常遷移
      const targetPath = url.split("?")[0];
      if (window.location.pathname === targetPath) {
        const params = new URLSearchParams(url.split("?")[1] || "");
        const messageId = params.get("m");
        if (messageId) {
          window.dispatchEvent(
            new CustomEvent("huddle:jumpToMessage", { detail: { messageId } })
          );
        }
      } else {
        window.location.href = url;
      }
    }
    notification.close();
  };

  // 5秒後に自動で閉じる
  setTimeout(() => notification.close(), 5000);
}

```


## Codex レビュー

- [優先度: 今すぐ] `onclick` 内の `targetPath = url.split("?")[0]` が、上の `new URL(url, window.location.origin).pathname` と判定ロジック不一致です。絶対URLや `#` 付きURLで同一チャンネル判定が壊れます。`const targetUrl = new URL(url, window.location.origin)` を先に作り、`targetUrl.pathname` / `targetUrl.searchParams.get("m")` を一貫して使ってください。

- [優先度: 今すぐ] `window.location.href = url` は、呼び出し元が外部URLを渡すと通知クリックで外部サイトへ遷移できます。通知URLがアプリ内リンク前提なら、`targetUrl.origin === window.location.origin` を検証し、遷移は `targetUrl.pathname + targetUrl.search + targetUrl.hash` に限定してください。

- [優先度: 後で] `new URL(...)` や `new Notification(...)` は環境や入力次第で例外になり得ます。通知ユーティリティなので、壊れたURLやブラウザ実装差でアプリ本体を落とさないよう `try/catch` で握って return するのが安全です。

- [優先度: 後で] テストは最低限、同一チャンネルかつフォーカス中は通知しない、別チャンネルは通知する、クリック時に同一パスなら `huddle:jumpToMessage` を dispatch、別パスなら同一オリジン内で遷移、外部URLは遷移しない、を追加してください。
